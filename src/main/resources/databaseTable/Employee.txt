create table employee (
   id TINYINT NOT NULL,
   name  VARCHAR(20) NOT NULL,
   sex   VARCHAR(10) NOT NULL,
   birthday   BIGINT  NOT NULL, 
   age       TINYINT  NOT NULL, 
   tel   VARCHAR(20) NOT NULL,
   identityId  VARCHAR(20) NOT NULL,
   storeId  INT,
   PRIMARY KEY(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
